def getDate():
    x = range(2,26,2)
    y = [20+random.random()*10 for _ in range(12)]
    return x,y

中文支持¶

Windows下字体在C:\Windows\Fonts,或运行->fonts；在Linux/mac下通过fc-list查看支持的字体，用fc-list :lang=zh查看支持的中文字体（冒号前有空格）

In [9]:
from matplotlib import font_manager

my_font = font_manager.FontProperties(fname='C:\Windows\Fonts\simsun.ttc')

x = range(120)
y = [random.randint(20,35) for _ in range(120)]
plt.figure(figsize=(16,9),dpi=100)
plt.plot(x,y)
_xticks = ["10点{}分".format(i) for i in range(60)]+["11点{}分".format(i) for i in range(60)] #这时坐标轴上的数字和x数据不同，需要指定替换位置和替换的值
plt.xticks(list(x)[::3],_xticks[::3],rotation=60,fontproperties=my_font) # rotation=90：x坐标轴刻度旋转90度
plt.show()

def addDescribe():
    plt.xlabel("时间",fontproperties=my_font)
    plt.ylabel("温度 单位（℃）",fontproperties=my_font)
    plt.title("一天气温变化图",fontproperties=my_font)
    
    plt.xticks(range(0,25))
def play():
    x,y = getDate()
    plt.plot(x,y)
    addDescribe()
play()
plt.show()
